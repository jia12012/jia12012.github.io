boolean isFirst = true;
for(int i=2 ; i<num;i++) {
    boolean isPrime = true;
    for(int j=2;j<i;j++) {
        if(i%j == 0) {
            isPrime = false;
        }
    }
    //此只會做一次後isPrime就會變false不再進去了
    if(isFirst && isPrime) {
        System.out.print(i);
        isFirst = false;
    }
    else if(isPrime)
        System.out.print(","+i);
}
sc.close();

Q23 School course system

In this question, you will design a school course system. We will give you part of the main method and some hints for the class and you should fill your code in the blank. We expect your code should generate the same output as ours.

  • First, in the beginning of the main method, we create two Course and two Student objects. Each course has a courseID and name. And each student has a studentID and name.
  • When we enroll a student to a course, we need to create an object of Enroll class, and the Enroll class will store the student and course and the grade of the student for the course.
  • Because each course can enroll many students, each course has an EnrollList to deal with this.
  • The EnrollList can store many Enroll and we define a static attribute in this class. This static attribute counts how many enrollments in total in this course system.
  • Note that your attributes should use a private access level modifier and use public getter to get some instance variables.


























public class Course {
    private String courseID;
    private String name;
    private EnrollList enrolls;

    public Course(String courseID, String name) {
        this.courseID = courseID;
        this.name = name;
        this.enrolls = new EnrollList();
    }

    public void enroll(Student student) {
        Enroll enroll = new Enroll(student, this);
        this.enrolls.addEnroll(enroll);
    }

    public void giveEveryoneGrade(int grade) {
        this.enrolls.giveEveryoneGrade(grade);
    }

    public void printSummary() {
        System.out.println("Course: " + this.name);
        this.enrolls.printSummary();
        System.out.println("-".repeat(40));
    }
}

public class Course {
    private int courseID;
    private String courseName;
    private int credits;
    private int capacity;
    private int enrolled;

    Course(int id, String name, int credits, int capacity) {
        this.courseID = id;
        this.courseName = name;
        this.credits = credits;
        this.capacity = capacity;
        this.enrolled = 0;
    }

    public boolean isFull() {
        return this.enrolled == this.capacity;
    }

    public void enroll() {
        ++this.enrolled;
    }

    public void drop() {
        --this.enrolled;
    }

    public String getInfo() {
        String s = "";
        s = String.format("%d %s %d %d/%d", this.courseID, this.courseName, this.credits, this.enrolled, this.capacity);
        return s;
    }

    public int getCourseID() {
        return this.courseID;
    }

    public int getCredits() {
        return this.credits;
    }

    public String getCourseName() {
        return this.courseName;
    }

    public int getCapacity() {
        return this.capacity;
    }
}


















































































public class Student {
    private int studentID;
    private String studentName;
    private ArrayList<Course> enrolledCourse;
    private int currentCredits;
    private int maxCredits;

    Student(int studentID, String name) {
        this.studentID = studentID;
        this.studentName = name;
        this.currentCredits = 0;
        this.maxCredits = 25;
        this.enrolledCourse = new ArrayList();
    }

    Student(int studentID, String name, int maxCredits) {
        this.studentID = studentID;
        this.studentName = name;
        this.maxCredits = maxCredits;
        this.currentCredits = 0;
        this.enrolledCourse = new ArrayList();
    }

    public Course getCourse(int courseID) {
        Iterator var3 = this.enrolledCourse.iterator();

        while(var3.hasNext()) {
            Course cou = (Course)var3.next();
            if (cou.getCourseID() == courseID) {
                return cou;
            }
        }

        return null;
    }

    public void enroll(Course cou) {
        if (cou.isFull()) {
            System.out.printf("Course %d is full\n", cou.getCourseID());
        } else if (this.currentCredits + cou.getCredits() > this.maxCredits) {
            System.out.printf("Student %d cannot enroll this course\n", this.studentID);
        } else {
            this.enrolledCourse.add(cou);
            cou.enroll();
            this.currentCredits += cou.getCredits();
        }

    }

    public void drop(int courseID) {
        if (this.getCourse(courseID) == null) {
            System.out.printf("Course %d not found in student %d\n", courseID, this.studentID);
        } else {
            this.currentCredits -= this.getCourse(courseID).getCredits();
            this.getCourse(courseID).drop();
            this.enrolledCourse.remove(this.getCourse(courseID));
        }

    }

    public void drop(Course cou) {
        this.drop(cou.getCourseID());
    }

    public String getInfo() {
        String s = "";
        String s_cou = "";

        Course cou;
        for(Iterator var4 = this.enrolledCourse.iterator();
            var4.hasNext();
            s_cou = s_cou + String.format("%d %s %d\n",
                                          cou.getCourseID(), cou.getCourseName(), cou.getCredits())) {
            cou = (Course)var4.next();
        }

        s = String.format("Student ID: %d\nName: %s\nCredits: %d/%d\nCourse list:\n%s", 
                          this.studentID, this.studentName, this.currentCredits, this.maxCredits, s_cou);
        return s;
    }
}

public class Employee {
	private int ID;
	private double baseSalary;
	private double totalSalary;
	private int sales;
	private String name;
	private String department;
	private BankAccount account;
    
	public Employee(int ID, String name, BankAccount account, String department, int baseSalary, int sales) {
		this.ID = ID;
		this.name = name;
		this.account = account;
		this.department = department;
		this.baseSalary = baseSalary;
		this.totalSalary = 0;
		this.sales = sales;
	}
	public String getDepartment() {
		return this.department;
	}
	public int getSales() {
		return this.sales;
	}
	public double getTotalSalary() {
		return this.totalSalary;
	}
	public void setSales(int sales) {
		this.sales = sales;
	}
	public void monthEnd() {
		double taxRate = 0.03;
		double salesBonus = 500;
		this.totalSalary = (this.baseSalary+(this.sales*salesBonus))*(1.0-taxRate);
        //把月末的錢存進去對應employee的account內
		this.account.deposit(this.totalSalary);
	}
	public String getInfo() {
		String s = "";
		s = String.format("ID:%d\nName:%s\nDepartment:%s\nTotal sales:%d\nTotal salary:%.0f\nAccount amount:%.0f", this.ID,
				this.name,this.department,this.sales,this.totalSalary,this.account.getBalance());
		return s;
	}
}
















import java.util.ArrayList;
public class Supervisor extends Employee{
	private ArrayList<Employee> subordinates = new ArrayList<Employee>();
    
	public Supervisor(int ID , String name, BankAccount account , String department, int baseSalary ,int sales, ArrayList<Employee> subordinates) {
		super(ID,name,account,department,baseSalary,sales);
		this.subordinates = subordinates;
		int subordinateSale = 0;
        //把Bob的所有下屬的sales加進去自己的sales
		for(Employee e: this.subordinates) {
			subordinateSale += e.getSales();
		}
		this.sales = sales+subordinateSale;
	}
}































public class BankAccount {
	private int ID;
	private double balance;
	public BankAccount(int ID, double balance) {
		this.ID = ID;
		this.balance = balance;
	}
	public int getID() {
		return ID;
	}
	public void setID(int ID) {
		this.ID = ID;
	}
	public void setBalance(double balance) {
		this.balance = balance;
	}
	public void deposit(double amount) {
		balance += amount;
	}
	public void withdraw(double amount) {
		if(balance < amount) {
			System.out.println("Your account does not have enough money");
		}else {
			balance -= amount;
		}
	}
	public double getBalance() {
		return this.balance;
	}
}
